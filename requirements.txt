# =====================================================================
# 萬用加密工具 v5.1 - Python 套件需求檔
# =====================================================================
# 安裝方式：
#   pip install -r requirements.txt
#
# 建議 Python 版本：Python 3.9 以上（含 3.9 ~ 3.12）
# =====================================================================

# ---- 圖形介面（必要）----
# 提供程式的美化版視窗介面（卡片、圓角按鈕、步驟精靈等）。
# 若未安裝，程式仍可啟動，但只會顯示「請安裝 customtkinter」的提示視窗。
customtkinter>=5.2.0

# ---- PDF 加密 / 讀寫（必要，PDF 相關功能都需要）----
# 用於：
#   1) PDF 原生加密（AES-256）
#   2) 「圖片轉加密 PDF」功能中，把圖片合併成 PDF 後再加密
pypdf>=4.0.0

# ---- 加密演算法後端（必要）----
# pypdf 在使用 AES-256 加密 PDF 時，實際運算依賴 cryptography 套件。
cryptography>=42.0.0

# ---- 圖片處理（必要，「圖片轉加密PDF」功能需要）----
# 用於讀取 jpg / png / bmp / tif / gif / webp 等圖片，並合併輸出成 PDF。
Pillow>=10.0.0

# =====================================================================
# 以下為「選用」套件
# =====================================================================

# ---- Office 原生加密（選用，僅限 Windows + 已安裝正版 Microsoft Office）----
# 用於：Word / Excel / PowerPoint 檔案的「Office 原生加密」功能
#       （透過 COM 呼叫本機安裝的 Office 應用程式進行加密）。
#
# 限制：
#   - 只能在 Windows 作業系統上安裝與運作
#   - 電腦必須已安裝正版 Microsoft Office（Word/Excel/PowerPoint）
#   - macOS / Linux 使用者請勿安裝，安裝也無法使用此功能，
#     程式會自動偵測並隱藏/停用該選項
pywin32>=306; sys_platform == "win32"
